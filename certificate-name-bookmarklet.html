var input = document.getElementById('certificateName') || document.querySelector('input[placeholder*="certificate"]') || document.querySelector('input[id*="certificate"]'); if (input) { input.value = 'SOUDHAME'; input.focus(); input.blur(); var event = new Event('input', {bubbles: true}); input.dispatchEvent(event); var changeEvent = new Event('change', {bubbles: true}); input.dispatchEvent(changeEvent); alert('✅ Certificate name set to SOUDHAME!'); } else { alert('❌ Certificate input field not found. Make sure you are on the Pro Dashboard page.'); }
